The Code

python — shopping_cart.py
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.sdk.resources import Resource
from opentelemetry.exporter.otlp.proto.http.trace_exporter import OTLPSpanExporter

def configure_tracer():
    # OTLP exporter — sends traces to the Collector
    exporter = OTLPSpanExporter(endpoint="http://localhost:4318/v1/traces")
    span_processor = BatchSpanProcessor(exporter)

    # Resource describes this service
    resources = Resource.create({
        "service.name": "Shopping Cart Service",
        "service.version": "0.2.0"
    })

    provider = TracerProvider(resource=resources)
    provider.add_span_processor(span_processor=span_processor)
    trace.set_tracer_provider(provider)
    return trace.get_tracer("shopping-cart", "0.2.0")

# Configure tracer at module level — before any functions that use it
tracer = configure_tracer()

def create_order():
    with tracer.start_as_current_span("Create Order"):
        print("creating Order")
        validate_order()
        reserve_inventory()

def validate_order():
    with tracer.start_as_current_span("Validate Order"):
        print("Validating Users Order")

def reserve_inventory():
    with tracer.start_as_current_span("Reserve Inventory"):
        print("Reserving Inventory")

if __name__ == "__main__":
    create_order()
🧠 Think about the user journey
This is a simplified version of what a real e-commerce trace looks like. In production, each span might cross service boundaries: the order service validates, the inventory service reserves stock, the payment service charges. With distributed tracing, you'd see the entire flow as one connected trace.

Project: Fitness Service Metrics

The fitness service project is a richer, more realistic application. It combines multiple metric instruments — counters, up-down counters, and histograms — in a service that handles workouts, meals, and image uploads with background processing.

What Makes This Interesting

🔢
Request Counting
Track total HTTP requests by route and method.
📸
Background Queue
Image uploads queue background tasks — the up-down counter shows queue depth.
⏱️
Latency Tracking
Random delays simulate real latency, measured by the histogram.

The Code

python — fitness.py
from flask import Flask, request
from opentelemetry.metrics import set_meter_provider, get_meter_provider
from opentelemetry.sdk.metrics import MeterProvider
from opentelemetry.sdk.resources import Resource
from opentelemetry.sdk.metrics.view import ExplicitBucketHistogramAggregation, View
from opentelemetry.exporter.prometheus import PrometheusMetricReader
from prometheus_client import start_http_server
import threading, time, random

def configure_meter():
    start_http_server(8000)  # Prometheus scrape endpoint
    reader = PrometheusMetricReader()
    buckets = [0.0, 20, 30, 60, 100, 250, 500, 1000]
    view = View(
        instrument_name="total_request_duration",
        aggregation=ExplicitBucketHistogramAggregation(boundaries=buckets)
    )
    provider = MeterProvider(resource=Resource.create(), metric_readers=[reader])
    set_meter_provider(provider)
    return get_meter_provider().get_meter(name="fitness-app", version="0.1.2")

meter = configure_meter()

# Counter — total HTTP requests
http_total_requests = meter.create_counter(
    "http_requests_total", unit="1",
    description="Total number of requests processed"
)

# Up-Down Counter — image processing queue depth
queue_length = meter.create_up_down_counter(
    "queue_length", unit="1",
    description="Number of image processing tasks in queue"
)

# Histogram — request duration
total_request_duration = meter.create_histogram(
    "total_request_duration", unit="ms",
    description="Duration of each HTTP request in milliseconds"
)

app = Flask(__name__)
task_queue = []

def process_image_task(task_id):
    """Simulates a background image processing job"""
    processing_time = 10
    print(f"[Task {task_id}] Processing for {processing_time} seconds...")
    time.sleep(processing_time)
    queue_length.add(-1, {"route": "/pics"})  # Queue shrinks when done
    print(f"[Task {task_id}] Done processing.")

@app.before_request
def before_request():
    request.start_time = time.time()
    simulated_delay = random.uniform(0.01, 1.0)
    time.sleep(simulated_delay)

@app.after_request
def after_request(response):
    request_latency = (time.time() - request.start_time) * 1000
    total_request_duration.record(request_latency)
    return response

# Image upload — triggers background processing
@app.post("/pics")
def upload_pic():
    http_total_requests.add(1, {"route": "/pics", "method": request.method})
    task_id = len(task_queue) + 1
    task_queue.append(task_id)
    queue_length.add(1, {"route": "/pics"})  # Queue grows
    thread = threading.Thread(target=process_image_task, args=(task_id,))
    thread.start()
    return f"Image {task_id} uploaded and queued for processing."

@app.get("/workouts")
def get_workouts():
    http_total_requests.add(1, {"route": "/workouts", "method": request.method})
    return "Get All workouts"

@app.post("/workouts")
def create_workouts():
    http_total_requests.add(1, {"route": "/workouts", "method": request.method})
    return "Create Workout"

@app.get("/meals")
def get_meals():
    http_total_requests.add(1, {"route": "/meals", "method": request.method})
    return "Get All Meals"

@app.post("/meals")
def create_meals():
    http_total_requests.add(1, {"route": "/meals", "method": request.method})
    return "Create Meal"

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
💡 Why the queue pattern matters
The image upload endpoint (/pics) is interesting because it queues a background task. The up-down counter tracks how many tasks are in the queue at any moment. In a real system, this metric tells you if your processing is keeping up with demand — if queue length keeps growing, you have a capacity problem.
